
```java
package com.njust.test;

class Base
{
    public int age = 1;
    
    public void say()
    {
        System.out.println("Base:" + age);
    }
}

class Child extends Base
{
    public int age = 100;
    
    public void say()
    {
        System.out.println("Child:" + age);
    }
}

public class ChildTest
{
    public static void main(String[] args)
    {
        Child child = new Child();
        Base base = child;
        
        System.out.println(base.age);
        System.out.println(child.age);
        
        base.say();
        child.say();
    }
}

```

```console
1
100
Child:100
Child:100
```
