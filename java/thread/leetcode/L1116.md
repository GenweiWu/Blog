## -1:利用1个object进行wait/notify  **超时**
```java
package org.njust.l1116;

import java.util.function.IntConsumer;

class ZeroEvenOdd {

    //0/1/2 -> 0/odd奇数/even偶数
    private int flag = 0;
    private int index = 1;

    private final Object object = new Object();

    private int n;

    public ZeroEvenOdd(int n) {
        this.n = n;
    }

    // printNumber.accept(x) outputs "x", where x is an integer.
    public void zero(IntConsumer printNumber) throws InterruptedException {
        while (index <= n) {
            synchronized (object) {
                while (flag != 0) {
                    object.wait();
                }
                //double check
                if (index > n) {
                    break;
                }

                printNumber.accept(0);
                flag = index % 2 != 0 ? 1 : 2;
                object.notifyAll();
            }
        }
    }

    public void even(IntConsumer printNumber) throws InterruptedException {
        while (index <= n) {
            synchronized (object) {
                while (flag != 2) {
                    object.wait();
                }
                //double check
                if (index > n) {
                    break;
                }

                printNumber.accept(index++);
                flag = 0;
                object.notifyAll();
            }
        }
    }

    public void odd(IntConsumer printNumber) throws InterruptedException {
        while (index <= n) {
            synchronized (object) {
                while (flag != 1) {
                    object.wait();
                }
                //double check
                if (index > n) {
                    break;
                }

                printNumber.accept(index++);
                flag = 0;
                object.notifyAll();
            }
        }
    }
}
```


## 1. 利用3个condition

> 没有使用一个全局的value，避免了value可能需要double check的问题

```java
// 1.仅仅依靠condition通知好像不靠谱，需要结合flag标识0/1/2来区分是打印ODD还是EVEN
    private transient int flag = 0;

    private Lock lock = new ReentrantLock();

    private Condition zeroTurn = lock.newCondition();

    private Condition oddTurn = lock.newCondition();

    private Condition evenTurn = lock.newCondition();

    private int n;

    public ZeroEvenOdd(int n) {
        this.n = n;
    }

    // printNumber.accept(x) outputs "x", where x is an integer.
    public void zero(IntConsumer printNumber) throws InterruptedException {

        lock.lock();
        try {

            for (int i = 1; i <= n; i++) {
                while (flag != 0) {
                    zeroTurn.await();
                }
                printNumber.accept(0);
                if (i % 2 == 0) {
                    flag = 2;
                    evenTurn.signal();
                } else {
                    flag = 1;
                    oddTurn.signal();
                }
            }
        } finally {
            lock.unlock();
        }
    }

    // 偶数
    public void even(IntConsumer printNumber) throws InterruptedException {

        lock.lock();
        try {

            for (int i = 2; i <= n; i = i + 2) {
                while (flag != 2) {
                    evenTurn.await();
                }
                printNumber.accept(i);
                flag = 0;
                zeroTurn.signal();
            }

        } finally {
            lock.unlock();
        }
    }

    // 奇数
    public void odd(IntConsumer printNumber) throws InterruptedException {

        lock.lock();
        try {
            for (int i = 1; i <= n; i = i + 2) {
                while (flag != 1) {
                    oddTurn.await();
                }

                printNumber.accept(i);
                flag = 0;
                zeroTurn.signal();
            }
        } finally {
            lock.unlock();
        }
    }
```



## 2. lock+condition*3

> 1.使用了公共变量value，涉及double check问题，否则可能出错，多打印
> 2.zero方法最后要多调用一次signal方法，来解决线程一直在wait的问题

```java
class ZeroEvenOdd {
    // 1.仅仅依靠condition通知好像不靠谱，需要结合flag标识0/1/2来区分是打印ODD还是EVEN
    private transient int flag = 0;

    private transient int value = 1;

    private Lock lock = new ReentrantLock();

    private Condition zeroTurn = lock.newCondition();

    private Condition oddTurn = lock.newCondition();

    private Condition evenTurn = lock.newCondition();

    private int n;

    public ZeroEvenOdd(int n) {
        this.n = n;
    }

    // printNumber.accept(x) outputs "x", where x is an integer.
    public void zero(IntConsumer printNumber) throws InterruptedException {

        lock.lock();
        try {

            //3.这里其实间接实现了double check
            while (value <= n) {
                if (flag != 0) {
                    zeroTurn.await();
                } else {
                    printNumber.accept(0);
                    if (value % 2 == 0) {
                        flag = 2;
                        evenTurn.signal();
                    } else {
                        flag = 1;
                        oddTurn.signal();
                    }
                }

            }

            // 2.再signal一次，否则程序一直不结束
            evenTurn.signal();
            oddTurn.signal();

        } finally {
            lock.unlock();
        }
    }

    // 偶数
    public void even(IntConsumer printNumber) throws InterruptedException {

        lock.lock();
        try {

            while (value <= n) {
                if (flag != 2) {
                    evenTurn.await();
                } else {
                    printNumber.accept(value++);
                    flag = 0;
                    zeroTurn.signal();
                }
            }

        } finally {
            lock.unlock();
        }
    }

    // 奇数
    public void odd(IntConsumer printNumber) throws InterruptedException {

        lock.lock();
        try {
            while (value <= n) {
                if (flag != 1) {
                    oddTurn.await();
                } else {
                    printNumber.accept(value++);
                    flag = 0;
                    zeroTurn.signal();
                }
            }

        } finally {
            lock.unlock();
        }
    }
}
```
